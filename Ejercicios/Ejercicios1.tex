\subsection{redistribucionDeLosFrutos}


\begin{proc}{redistribucionDeLosFrutos}{\In recursos : \TLista{\float}, \In cooperan : \TLista{\bool}}{\TLista{\float}}
	\requiere{|recursos| > 0 \yLuego |recursos| = |cooperan| }
	\asegura {\\
 |res| = |cooperan| \yLuego \\ 
 \paraTodo[unalinea]{i}{\ent} {0 \leq i \leq |res|  \yLuego cooperan[i] = True \implicaLuego res[i] = \frac{fondo(recursos)}{|cooperan|} }  \yLuego \\ \paraTodo[unalinea]{i}{\ent} {0 \leq i \leq |res| \yLuego cooperan[i] = False \implicaLuego res[i] = recursos[i] + \frac{fondo(recursos)}{|cooperan|}} \\
 }

\end{proc}

\vspace{0.50cm}

\aux{fondo}{in recursos : {\TLista{\ent}} }{\ent} {
\sum\limits_{i=0} \limits^{|recursos|-1} 
( {\IfThenElse{cooperan[i] \hspace{0.15cm} = \hspace{0.15cm} True}{recursos[i]}{0}} )}
